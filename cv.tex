\newcommand{\profilesection}[1]{%
    \vspace{\baselineskip}% 
    {\par\huge\textcolor{highlight}{#1}
     \hspace{5pt}\textcolor{highlight}{\sout{\hfill}}}\par% 
    \vspace{5pt}% 
}

\newcommand{\profilesectionbox}[1]{%
    \vspace{\baselineskip}% 
    \tikz \node [shape=rectangle, fill=white, background, text width=\linewidth, align=left, text=highlight] {
        {\par\huge\textcolor{highlight}{#1}\hspace{5pt}\textcolor{highlight}{\sout{\hfill}}}
    };
}

\newcommand\skills[1]{ 
    \foreach [count=\i] \x/\y in {#1}
    {    
        \begin{tikzpicture}[]
            \draw[color=lightgray, line width=0.8ex] (0,0) arc (0:360:2.2ex) node[align=left,  anchor=west, color=black] { \x };
            \draw[color=highlight, line width=0.8ex] (0,0) arc (0:3.6*(\y)):2.2ex);
            \node(box) at (-2.2ex, 0cm) {\footnotesize{\y\%}};
        \end{tikzpicture}
        \par
    }
}

\newcommand\skillsline[1]{ 
    \foreach [count=\i] \x/\y in {#1}
    {
        \hspace*{5pt}{\x}\par
        \vspace{2pt}% 
        \begin{tikzpicture}
            \draw[fill=white, lightgray](0, 0) rectangle (\linewidth, 0.3);
            \draw[fill=white, highlight](0, 0) rectangle (0.1667*\y\linewidth, 0.3);
        \end{tikzpicture}
    }
}
